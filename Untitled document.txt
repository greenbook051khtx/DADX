Title.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
Title.Font = Enum.Font.SourceSansBold
Title.TextSize = 18
Title.Parent = MainFrame


local _G = _G or {}
_G.ChestFarm = false
_G.FruitFarm = false


local function createToggleButton(text, position, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0, 180, 0, 40)
    btn.Position = position
    btn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    btn.Text = text .. " : OFF"
    btn.TextColor3 = Color3.fromRGB(255, 100, 100)
    btn.Font = Enum.Font.SourceSans
    btn.TextSize = 16
    btn.Parent = MainFrame
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = btn
    
    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        if active then
            btn.BackgroundColor3 = Color3.fromRGB(50, 150, 50)
            btn.TextColor3 = Color3.fromRGB(255, 255, 255)
            btn.Text = text .. " : ON"
        else
            btn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
            btn.TextColor3 = Color3.fromRGB(255, 100, 100)
            btn.Text = text .. " : OFF"
        end
        callback(active)
    end)
end


ToggleBtn.MouseButton1Click:Connect(function()
    MainFrame.Visible = not MainFrame.Visible
end)


game:GetService("StarterGui"):SetCore("SendNotification", {
    Title = "BloxHub Mobile",
    Text = "Menu loaded successfully!",
    Duration = 3
})


createToggleButton("Auto Farm Chest", UDim2.new(0, 20, 0, 60), function(state)
    _G.ChestFarm = state
end)


task.spawn(function()
    while task.wait(1) do
        if _G.ChestFarm then
            local char = player.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                for _, obj in pairs(workspace:GetDescendants()) do
                    if _G.ChestFarm and obj.Name:find("Chest") then
                        local targetCFrame = nil
                        if obj:IsA("Model") and obj.PrimaryPart then
                            targetCFrame = obj.PrimaryPart.CFrame
                        elseif obj:IsA("Part") or obj:IsA("MeshPart") then
                            targetCFrame = obj.CFrame
                        end
                        
                        if targetCFrame then
                            char.HumanoidRootPart.CFrame = targetCFrame
                            task.wait(1.5)
                        end
                    end
                end
            end
        end
    end
end)


createToggleButton("Auto Collect Fruit", UDim2.new(0, 20, 0, 110), function(state)
    _G.FruitFarm = state
end)


workspace.ChildAdded:Connect(function(child)
    if child.Name:find("Fruit") then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "FRUIT DETECTED!",
            Text = "A " .. child.Name .. " has spawned!",
            Duration = 10
        })
    end
end)


task.spawn(function()
    while task.wait(2) do
        if _G.FruitFarm then
            local char = player.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                for _, obj in pairs(workspace:GetChildren()) do
                    if _G.FruitFarm and (obj.Name:find("Fruit") or obj:FindFirstChild("Handle")) and obj:IsA("Tool") then
                        local handle = obj:FindFirstChild("Handle") or obj
                        char.HumanoidRootPart.CFrame = handle.CFrame
                        task.wait(2)
                    end
                end
            end
        end
    end
end)
